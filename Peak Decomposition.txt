namespace statusbar {
    class StaminaBar {
        -int maxHealth
        +int currentHealth get; private set;
        +List~int~ statusValues
        +Dictionary~String, int~ StatusDictionary;

        +GetStatus(StatusName) Statuses
        +CheckForDie() bool
        +UpdateHunger()
        -Update()
    }

        +float KnockoutTimer
